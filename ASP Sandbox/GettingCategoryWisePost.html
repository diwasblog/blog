// @nuget: HtmlAgilityPack

using System;
using System.Xml;
using HtmlAgilityPack;using System;
using System.Collections.Generic;
					
public class Program
{
	public static void Main()
	{
		var s=new List();
  			s.Add(new Category{Id=1,Name="Cat1"});
  			s.Add(new Category{Id=2,Name="Cat2"});
			s.Add(new Category{Id=3,Name="Cat3"});
		
		var ss=new List();
		 ss.Add(new Post{Id=1,Name="Diwas",CategoryId=1});
		ss.Add(new Post{Id=2,Name="Hari",CategoryId=2});
		ss.Add(new Post{Id=3,Name="Ram",CategoryId=3});
		ss.Add(new Post{Id=4,Name="Shyam",CategoryId=1});
		
		foreach(var item in s)
		{
			Console.WriteLine("===================================Category:"+item.Name+"=================================");
			foreach(var items in ss)
			{
				if (item.Id==items.CategoryId)
				{
					Console.WriteLine(items.Name);	
				}
			}
			
		}
	}
}

public class Category
{
	public int Id{get;set;}
	public string Name{get;set;}
}

public class Post
{
	public int Id{get;set;}
	public string Name{get;set;}
	public int CategoryId{get;set;}
}
