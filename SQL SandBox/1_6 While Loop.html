DECLARE	@Counter INT
DECLARE @MaxOscars INT
DECLARE @NumFilms INT

SET @MaxOscars=(Select MAX(OscarWins) from Film)
SET @Counter=0

WHILE @Counter <=@MaxOscars
	BEGIN
		SET @NumFilms=(Select Count(*) from Film where OscarWins=@Counter)

		If @NumFilms=0 BREAK;
		
		PRINT CAST(@NumFilms AS VARCHAR(3)) +' films have won' +
			 CAST(@Counter AS VARCHAR(2))+'Oscars'
			 
		SET @Counter=@Counter+1
	END
