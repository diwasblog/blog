Create Database

    
      CREATE DATABASE database_name   //database name must be unique

      DROP DATABASE database_name

RENAME Database

    
      ALTER DATABASE old_name MODIFY NAME=new_name

To Select /Use DATABASE

    
      USE DATABASE database_name

Create Table

    
     CREATE TABLE Employee
     (
       EmployeeId int,
       FirstName varchar(50),
       LastName varchar(50),
       Email varchar(255) NOT NULL,
       Address varchar(255),
       primary key(id)
     );

Drop Table

    
     DROP TABLE Employee

Check TAble Exists or not

    
     DESC Employee

Delete Table

    
     DELETE FROM Employee;  //delete all row of table
     DELETE FROM Employee [where condition]  //delete based on condition
