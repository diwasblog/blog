Generate Input Tag Dynamically and add to Body

<body> ... </body> <script> let newElement = document.createElement('input'); document.body.appendChild(newElement); </script>

Generate Image Tag Dynamically and add to Container Div

<div id="container"></div> <script> var url = 'https://avatarfiles.alphacoders.com/822/82242.png'; var image = new Image(); image.src = url; document.getElementById('container').appendChild(image); </script>

Create List Dynamically

<div id='container'></div> <script> var arr = ['alpha', 'bravo', 'charlie', 'delta', 'echo']; var cont = document.getElementById('container'); // create ul element and set the attributes. var ul = document.createElement('ul'); ul.setAttribute('style', 'padding: 0; margin: 0;'); ul.setAttribute('id', 'theList'); for (i = 0; i <= arr.length - 1; i++) { var li = document.createElement('li'); // create li element. li.innerHTML = arr[i]; // assigning text to li using array value. li.setAttribute('style', 'display: block;'); // remove the bullets. ul.appendChild(li); // append li to ul. } cont.appendChild(ul); // add list to the container. </script>
//Final Generated Output <div id="container"> <ul style="padding: 0; margin: 0;" id="theList"> <li style="display: block;">alpha</li> <li style="display: block;">bravo</li> <li style="display: block;">charlie</li> <li style="display: block;">delta</li> <li style="display: block;">echo</li> </ul> </div>
