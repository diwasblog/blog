Object.keys() extracts keys from a given object

================================================================================================
Input : var check = ['x', 'y', 'z']; console.log(Object.keys(check)); Output : ['0', '1', '2']
Input : var check = ['x', 'y', 'z']; console.log(Object.values(check)); Output : Array ["x", "y", "z"]

==========================================================================================
================================================================================================
var object = { 0: 'x', 1: 'y', 2: 'z' }; console.log(Object.keys(object)); Output : ['0', '1', '2']
Input : var object = { 0: 'x', 1: 'y', 2: 'z' }; console.log(Object.values(object)); Output : Array ["x", "y", "z"]

==========================================================================================
Input : var object = { 70: 'x', 21: 'y', 35: 'z' }; console.log(Object.keys(object)); Output : ['21', '35', '70']
Input : var object = { 70: 'x', 21: 'y', 35: 'z' }; console.log(Object.values(object)); Output : Array ["y", "z", "x"]

====================================================================================== var nameObj = {first: "Diwas", last: "Poudel"}; console.log(Object.keys(nameObj)); // Outputs: // ["first", "last"]

Object.Keys with foreach

var nameObj = {first: "Diwas", last: "Diwas"}; Object.keys(nameObj).forEach(function(key) { console.log(key + ': ' + nameObj[key]); }); // Outputs: // first: Diwas // last: Poudel =====================================================================================================================
