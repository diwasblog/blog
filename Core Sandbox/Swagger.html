1. Create Web api project. By default we have test api url
  2. Then install : Swashbuckle.AspNetCore
  3. In startup.cs add this: 

       public void ConfigureServices(IServiceCollection services)
        {
            services.AddMvc().SetCompatibilityVersion(CompatibilityVersion.Version_2_2);

            services.AddSwaggerGen(c =>
            {
                c.SwaggerDoc("v1", new Info { Title = "My API", Version = "v1" });
            });

        }

 

public void Configure(IApplicationBuilder app, IHostingEnvironment env)
        {


            app.UseSwagger();


            app.UseSwaggerUI(c=> {
                c.SwaggerEndpoint("/swagger/v1/swagger.json", "My API version 1");
            });

     ......
}

4. Then run the project , goto localhost:xxxx/swagger/index.html
