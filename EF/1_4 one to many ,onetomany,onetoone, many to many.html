[HttpPost] [Route("repomethod/insert")] public async Task<IActionResult> InsertRepository(RepositoryMethod repoMethod) { try { repoMethod.Name = repoMethod.Name.TrimStart().TrimEnd(); repoMethod.AdminName = User.Identity.Name.ToString(); repoMethod.CreatedDate = DateTime.UtcNow.AddMinutes(345); _theContext.RepositoryMethod.Add(repoMethod); var result = await _theContext.SaveChangesAsync(); if (result > 0) { return Ok("The Repository Method has been inserted!"); } else { throw new Exception("Can't insert the Repository right now!"); } } catch (Exception ex) { return BadRequest(ex.Message); } }

Inserting Service Method ( with comma separated servicerepository Method)

[HttpPost] [Route("servicemethod/insert")] public async Task<IActionResult> InsertService(ServiceMethod serviceMethod) { try { serviceMethod.Name = serviceMethod.Name.TrimStart().TrimEnd(); serviceMethod.AdminName = User.Identity.Name.ToString(); serviceMethod.CreatedDate = DateTime.UtcNow.AddMinutes(345); _theContext.ServiceMethod.Add(serviceMethod); await _theContext.SaveChangesAsync(); var lst = GetRepositoryMethodsFromIds(serviceMethod.CommaSeparatedRepoMethodIds); foreach (var item in lst) { ServiceMethodRepositoryMethod obj = new ServiceMethodRepositoryMethod(); obj.RepositoryMethodId = item.Id; obj.ServiceMethodId = serviceMethod.Id; _theContext.ServiceMethodRepositoryMethod.Add(obj); } var result = await _theContext.SaveChangesAsync(); if (result > 0) { return Ok("The Service Method has been inserted!"); } else { throw new Exception("Can't insert the Service Method right now!"); } } catch (Exception ex) { return BadRequest(ex.Message); } } [NonAction] private List<RepositoryMethod> GetRepositoryMethodsFromIds(string ids) { List<RepositoryMethod> theList = new List<RepositoryMethod>(); string[] theArray = ids.Split(','); for (int i = 0; i < theArray.Length - 1; i++) { theList.Add(_theContext.RepositoryMethod.FirstOrDefault(x=> x.Id == Convert.ToInt32(theArray[i]))); } return theList; }

Update Service Related

[HttpPost] [Route("servicemethod/edit")] public async Task<IActionResult> EditService(ServiceMethod serviceMethod) { ServiceNewEditPageVM obj = new ServiceNewEditPageVM(); try { var oldService = _theContext.ServiceMethod.FirstOrDefault(r => r.Id == serviceMethod.Id); oldService.Name = serviceMethod.Name.TrimStart().TrimEnd(); oldService.AdminName = User.Identity.Name.ToString(); oldService.ModifiedDate = DateTime.UtcNow.AddMinutes(345); oldService.IsActive = serviceMethod.IsActive; oldService.ParameterCount = serviceMethod.ParameterCount; oldService.AreaId = serviceMethod.AreaId; oldService.ServiceId = serviceMethod.ServiceId; await _theContext.SaveChangesAsync(); var CommaSeparatedServiceRepoIds = GetRepositoryMethodsFromIds(serviceMethod.CommaSeparatedRepoMethodIds); var theObjects = _theContext.ServiceMethodRepositoryMethod.Include(x => x.RepositoryMethod).Where(x => x.ServiceMethod.Id == serviceMethod.Id); //deleting already existing ids of servicerepomethod foreach (var item in theObjects) { if (!CommaSeparatedServiceRepoIds.Any(x => x.Id == item.RepositoryMethod.Id)) { _theContext.ServiceMethodRepositoryMethod.Remove(item); } } foreach (var item in CommaSeparatedServiceRepoIds) { if (!_theContext.ServiceMethodRepositoryMethod.Where(x => x.ServiceMethod.Id == serviceMethod.Id).Any(x => x.RepositoryMethod.Id == item.Id)) { var _temp = new ServiceMethodRepositoryMethod() { ServiceMethod = serviceMethod, RepositoryMethod = item, }; _theContext.ServiceMethodRepositoryMethod.Add(_temp); } else { var oldServiceRepoMethod = _theContext.ServiceMethodRepositoryMethod.Where(x => x.ServiceMethod.Id == serviceMethod.Id).Where(y => y.RepositoryMethod.Id == item.Id); } } await _theContext.SaveChangesAsync(); return Ok("The Service Method has been Updated!"); } catch (Exception ex) { return BadRequest(ex.Message); } }

Detele

[HttpPost] [Route("servicemethod/delete")] public async Task<IActionResult> DeleteServiceMethod(ServiceMethod serviceMethod) { try { var toDeleteServiceMethod = _theContext.ServiceMethod.Where(r => r.Id == serviceMethod.Id).FirstOrDefault(); var toDeleteServiceRepoMethod = _theContext.ServiceMethodRepositoryMethod.Where(r => r.ServiceMethodId == serviceMethod.Id).ToList(); foreach (var item in toDeleteServiceRepoMethod) { _theContext.Remove(item); } _theContext.Remove(toDeleteServiceMethod); var result = await _theContext.SaveChangesAsync(); if (result > 0) { return Ok("Repository Method has been deleted!"); } else { throw new Exception("Can't Delete the Method now!"); } } catch (Exception ex) { return BadRequest(ex.Message); } }
