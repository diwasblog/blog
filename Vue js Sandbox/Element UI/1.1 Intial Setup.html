For ElementUI Component Details

https://element.eleme.io/#/en-US/component/installation ====================================================================================

When you install vue cli 3 then it automatically comes with element ui


Setup ElementUi

npm i element-ui -S

In main.js

For Fully Importing import Vue from 'vue'; import ElementUI from 'element-ui'; import 'element-ui/lib/theme-chalk/index.css'; import App from './App.vue'; Vue.use(ElementUI); new Vue({ el: '#app', render: h => h(App) });
For partial importing (only button and select) import Vue from 'vue'; import { Button, Select } from 'element-ui'; //only importing Button and select import App from './App.vue'; Vue.component(Button.name, Button); Vue.component(Select.name, Select); /* or * Vue.use(Button) * Vue.use(Select) */ new Vue({ el: '#app', render: h => h(App) });
