<div id="app"> <ul> <li v-for="name in names" v-text="name"></li> </ul> <input type="text" v-model="newName" id="input"> <button @click="addName">Add Name</button> </div> <script> const vm=new Vue({ el:'#app', data:{ newName:'', names:["amar","badal","cital","diwas"] }, methods:{ addName(){ this.names.push(this.newName); } } }) </script>
