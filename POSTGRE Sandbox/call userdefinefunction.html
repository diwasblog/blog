USE OF OUT AND IN and userdefined function 

                    
                            CREATE FUNCTION minmax(a integer, b integer, c integer, OUT max integer,OUT min integer)
                            AS $$
                            BEGIN
                            max = greatest(a,b,c);
                            min = least(a,b,c);
                            END;
                            $$
                            language plpgsql
                            
                            //no need to return because OUT max, OUT min auto returns
                           
                     
                
                        
                                select minmax(20,30,40)

                                //Output:
                                (40,20)

USE OF OUTIN and userdefined function 

                    
                            CREATE FUNCTION minmax(a integer, b integer, c integer, OUT max integer,OUT min integer)
                            CREATE FUNCTION cal_sq(INOUT a numeric)
                            AS $$
                            BEGIN
                            a = a*a;
                            END
                            $$
                            LANGUAGE PLPGSQL
                            
                            
                           
                     
                
                        
                                select * FROM cal_sq(5)

                                //Output:
                                25
