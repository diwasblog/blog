CREATE OR REPLACE FUNCTION fn_summary_chart_gals_dashboard()
RETURNS TABLE(count_gals bigint,count_mentor bigint,avg_gals numeric) AS $$

	DECLARE
		total_member bigint;
		total_gals bigint;
		avg_gals numeric;
	BEGIN


		select count(m.group_id) into total_member from members m
		join work_flow wf on wf.module_member_id = m.group_id and m.province_id =2;

		select count(m.group_id) into count_gals from members m where m.is_gals = true and m.province_id =2;

		
		select count(m.group_id) into count_mentor from members m where m.is_gals_mentor = true and m.province_id =2;


		avg_gals := (count_gals / total_member::float) * 100;
		
		

		RETURN QUERY SELECT count_gals,count_mentor,avg_gals;
	END;
$$ LANGUAGE plpgsql;

select * from fn_summary_chart_gals_dashboard()
