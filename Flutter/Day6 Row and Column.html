Using Container

If include body with container then container will wrap whole page and gets whole page height and width so here color effect is applied to whole body 
            
                import 'package:flutter/material.dart';

                void main() => runApp(MaterialApp(
                  home:Home(),
                ));
                
                class Home extends StatelessWidget{
                  @override
                  Widget build(BuildContext context) {
                    // TODO: implement build
                   return Scaffold(
                     appBar: AppBar(
                       title:Text("First Flutter Apps"),
                       centerTitle: true,
                       backgroundColor: Colors.red[600],
                     ),
                     body:Row(
                       children: [
                         Text("First"),
                         FlatButton(
                           onPressed: (){},
                           color:Colors.amber,
                           child: Text("Second"),
                         ),
                         Container(
                           color:Colors.cyan,
                           padding:EdgeInsets.all(30.0),
                           child:Text("Inside Container"),
                         )
                       ],
                     ),
                     floatingActionButton: FloatingActionButton(
                       child: Text("Save"),
                       backgroundColor: Colors.green[600],
                     ),
                   );
                  }
                  
                }
