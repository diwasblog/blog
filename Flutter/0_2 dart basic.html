Dart List

        
        List names = ["Diwas","Yogesh","Dipesh"];
        print(names);            //  [Diwas,Yogesh,Dipesh];
        
        names.add("Hari");       // [Diwas,Yogesh,Dipesh,Hari];
        print(names);

        names.add(30);
        print(names);           // [Diwas,Yogesh,Dipesh,Hari,30];
        Here, we can see List is accepting any types of variables.
