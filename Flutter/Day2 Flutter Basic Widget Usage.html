Applying Color And Fonts

Step 6 ) Applying color to appbar and button and font to body text 
            

             
                import 'package:flutter/material.dart';

                void main() => runApp(MaterialApp(
                  home: Scaffold(
                    appBar: AppBar(
                      title:Text("First Flutter Apps"),
                      centerTitle: true,
                      backgroundColor: Colors.red[600],
                    ),
                    body:Center(
                      child:Text(
                        "This is body",
                        style: TextStyle(
                          fontSize:20.0,
                          fontWeight: FontWeight.bold,
                          letterSpacing: 2.0,
                          color:Colors.grey[600],
                        ),
                      ),
                    ),
                    floatingActionButton: FloatingActionButton(
                      child: Text("Save"),
                      backgroundColor: Colors.red[600],
                    ),
                  ),
                ));
